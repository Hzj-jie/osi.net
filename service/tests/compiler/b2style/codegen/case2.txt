
bool is_even(int i) {
  return b2style::equal(b2style::mod(i, 2), 0);
}

void main() {
  for (int i = 0; b2style::less_than(i, 100); i = b2style::self_inc(i)) {
    int x;
    x = b2style::current_ms();
    if (is_even(x)) {
      x = b2style::self_dec(x);
    }
    b2style::bool_std_out(is_even(x));
  }
}
