
#include <b2style.h>
#include <testing.h>
#include <b2style/delegates.h>

class C {
  b2style::function<int, int> inc;
  int base;

  int my_inc(int x) {
    return x + this.base;
  }

  void construct(int base) {
    this.inc =  my_inc;
	this.base = base;
  }
};

void main() {
  C c(1);
  b2style::testing::assert_equal<int>(c.inc(1), 2);
  b2style::testing::finished();
}
