
#include <b2style.h>
#include <testing.h>
#include <b2style/delegates.h>

class C {
  b2style::function<int, int> inc;

  int my_inc(int x) {
    return x + 1;
  }

  void construct() {
    this.inc =  my_inc;
  }
};

void main() {
  C c[1];
  c[0].construct();
  b2style::testing::assert_equal<int>(c[0].inc(1), 2);
  b2style::testing::finished();
}
