
#include <b2style.h>
#include <testing.h>

template <T>
class array {
  T a;

  void construct(int size) {
    T x[size];
	this.a = x;
	logic "undefine x";
  }

  void destruct() {
    logic "dealloc_heap this.a";
  }

  T get(int index) {
    return this.a[index];
  }

  void set(int index, T v) {
    this.a[index] = v;
  }
};

void main() {
  array<int> a(100);

  for (int i = 0; i < 100; i++) {
    a.set(i, i + 1);
  }
  for (int i = 0; i < 100; i++) {
    b2style::testing::assert_equal<int>(a.get(i), i + 1);
  }
  b2style::testing::finished();
}
